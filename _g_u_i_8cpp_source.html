446string addString(string baseString, string newString, int x1, int y1) {
447 // Split the base string and new string into lines
448 vector<string> baseLines;
449 vector<string> newLines;
450 stringstream ssBase(baseString), ssNew(newString);
451 string line;
452
453 while (getline(ssBase, line))
454 {
455 baseLines.push_back(line);
456 }
457
458 while (getline(ssNew, line))
459 {
460 newLines.push_back(line);
461 }
462
463 // Ensure the replacement does not go out of bounds
464 for (int i = 0; i < newLines.size(); i++)
465 {
466 int targetLine = y1 + i;
467 if (targetLine < baseLines.size())
468 {
469 string &baseLine = baseLines[targetLine];
470 string &newLine = newLines[i];
471
472 if (x1 < baseLine.size())
473 {
474 // Replace the part of the line starting from x1 with the new string
475 baseLine.replace(x1, newLine.size(), newLine);
476 }
477 else
478 {
479 // If x1 is outside the base line, just append the new string
480 baseLine.append(string(x1 - baseLine.size(), ' ') + newLine);
481 }
482 }
483 }
484
485 // Recombine the base lines into a single string
486 string result;
487 for (const string &baseLine : baseLines)
488 {
489 result += baseLine + "\n";
490 }
491
492 return result;
493}

598std::string GUI::getRandomPlayerName() {
599 std::string startPath = GUI::getFilePathStart();
600
601 // Get the contents of the adjectives.txt file
602 std::string adjectives = getFileContents(startPath + "/images/adjectives.txt");
603
604 // Split the adjective string into lines
605 std::vector<std::string> adjectiveList;
606 std::stringstream ssAdjectives(adjectives);
607 std::string adjective;
608 while (std::getline(ssAdjectives, adjective)) {
609 adjectiveList.push_back(adjective);
610 }
611
612 // Get the contents of the animals.txt file
613 std::string animals = getFileContents(startPath + "/images/animals.txt");
614
615 // Split the animals string into lines
616 std::vector<std::string> animalList;
617 std::stringstream ssAnimals(animals);
618 std::string animal;
619 while (std::getline(ssAnimals, animal)) {
620 animalList.push_back(animal);
621 }
622
623 // Initialize random number generator statically so it's only seeded once
624 auto seed = std::chrono::system_clock::now().time_since_epoch().count();
625 std::mt19937 gen(static_cast<unsigned long>(seed)); // Seeding the generator
626
627
628 // Randomly pick an adjective and animal from the lists
629 std::uniform_int_distribution<> distrAdjective(0, adjectiveList.size() - 1);
630 std::string randomAdjective = adjectiveList[distrAdjective(gen)];
631
632 std::uniform_int_distribution<> distrAnimal(0, animalList.size() - 1);
633 std::string randomAnimal = animalList[distrAnimal(gen)];
634
635 return randomAdjective + randomAnimal;
636}
