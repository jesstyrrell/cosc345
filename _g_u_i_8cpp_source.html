354

360  // Ensure the replacement does not go out of bounds

363  int targetLine = y1 + i;

365  {

367  string &newLine = newLines[i];

371  // Replace the part of the line starting from x1 with the new string
372  baseLine.replace(x1, newLine.size(), newLine);

376  // If x1 is outside the base line, just append the new string
377  baseLine.append(string(x1 - baseLine.size(), ' ') + newLine);

379  }
380  }
381 
382  // Recombine the base lines into a single string
383  string result;
