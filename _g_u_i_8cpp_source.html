447 string addString(string baseString, string newString, int x1, int y1) {
448  // Split the base string and new string into lines
449  vector<string> baseLines;
450  vector<string> newLines;
451  stringstream ssBase(baseString), ssNew(newString);
452  string line;
453 
454  while (getline(ssBase, line))
455  {
456  baseLines.push_back(line);
457  }
458 
459  while (getline(ssNew, line))
460  {
461  newLines.push_back(line);
462  }
463 
464  // Ensure the replacement does not go out of bounds
465  for (int i = 0; i < newLines.size(); i++)
466  {
467  int targetLine = y1 + i;
468  if (targetLine < baseLines.size())
469  {
470  string &baseLine = baseLines[targetLine];
471  string &newLine = newLines[i];
472 
473  if (x1 < baseLine.size())
474  {
475  // Replace the part of the line starting from x1 with the new string
476  baseLine.replace(x1, newLine.size(), newLine);
477  }
478  else
479  {
480  // If x1 is outside the base line, just append the new string
481  baseLine.append(string(x1 - baseLine.size(), ' ') + newLine);
482  }
483  }
484  }
485 
486  // Recombine the base lines into a single string
487  string result;
488  for (const string &baseLine : baseLines)
489  {
490  result += baseLine + "\n";
491  }
492 
493  return result;
494 }
