104vector<float> EquityCalculator::evaluateHand(const vector<Card>& hand, const vector<Card>& communityCards, Deck deck, int numPlayers) {
105
106 int iterations = 0;
107 double timeToCalculate = 1 * 1000000000; // 1 second in nano seconds
108 auto start = std::chrono::high_resolution_clock::now();
109
110 int wins = 0;
111 int draws = 0;
112 int losses = 0;
113 while (true) {
114
115 auto now = std::chrono::high_resolution_clock::now();
116 std::chrono::duration<double, std::nano> elapsed = now - start;
117 double num = elapsed.count();
118 if (elapsed.count() > timeToCalculate) { break; }
119
120 iterations++;
121 bool lose = false;
122 bool draw = false;
123 Deck deckCopy(deck);
124 deckCopy.shuffle();
125 vector<Card> communityCardsFull(communityCards);
126 while (communityCardsFull.size() < 5) {
127 communityCardsFull.push_back(deckCopy.deal());
128 }
129 for (int player = 1; player < numPlayers; player++) {
130 if (lose) { continue; }
131 // create random hand for opponent
132 vector<Card> otherHand;
133 for (int i = 0; i < 2; i++) { otherHand.push_back(deckCopy.deal()); }
134 // check if we beat opponent's hand
135 int outcome = handEvaluator.compareHands(hand, otherHand, communityCardsFull);
136 lose = outcome == LOSE;
137 if (!draw) {
138 draw = outcome == DRAW;
139 }
140 }
141 if (lose) { losses++; continue; }
142 if (draw) { draws++; continue; }
143 wins++;
144 }
145
146 return { 1.0f * wins / iterations, 1.0f * draws / iterations, 1.0f * (iterations - (wins + draws)) / iterations };
147}
